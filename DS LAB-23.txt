#include<stdio.h>
#define V 5
#define INF 999

int min(int dist[],int visited[]){
    int m=INF,idx;
    for(int i=0;i<V;i++)
        if(!visited[i] && dist[i]<m){
            m=dist[i]; idx=i;
        }
    return idx;
}

int main(){
    int g[V][V]={
        {0,10,0,5,0},
        {0,0,1,2,0},
        {0,0,0,0,4},
        {0,3,9,0,2},
        {7,0,6,0,0}
    };

    int dist[V],visited[V]={0};

    for(int i=0;i<V;i++) dist[i]=INF;
    dist[0]=0;

    for(int i=0;i<V-1;i++){
        int u=min(dist,visited);
        visited[u]=1;

        for(int v=0;v<V;v++)
            if(g[u][v] && dist[u]+g[u][v]<dist[v])
                dist[v]=dist[u]+g[u][v];
    }

    for(int i=0;i<V;i++)
        printf("0 -> %d = %d\n",i,dist[i]);
}
