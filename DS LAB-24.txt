#include<stdio.h>
#define V 5
#define INF 999

int main(){
    int g[V][V]={
        {0,2,0,6,0},
        {2,0,3,8,5},
        {0,3,0,0,7},
        {6,8,0,0,9},
        {0,5,7,9,0}
    };

    int selected[V]={0};
    selected[0]=1;

    for(int e=0;e<V-1;e++){
        int min=INF,x=0,y=0;
        for(int i=0;i<V;i++)
            if(selected[i])
                for(int j=0;j<V;j++)
                    if(!selected[j] && g[i][j] && g[i][j]<min){
                        min=g[i][j];
                        x=i; y=j;
                    }
        printf("%d - %d = %d\n",x,y,min);
        selected[y]=1;
    }
}
