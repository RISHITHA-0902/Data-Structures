#include<stdio.h>

void quick(int a[],int low,int high){
    if(low<high){
        int i=low,j=high,pivot=a[low],temp;
        while(i<j){
            while(a[i]<=pivot) i++;
            while(a[j]>pivot) j--;
            if(i<j){
                temp=a[i]; a[i]=a[j]; a[j]=temp;
            }
        }
        temp=a[low]; a[low]=a[j]; a[j]=temp;
        quick(a,low,j-1);
        quick(a,j+1,high);
    }
}

int main(){
    int a[]={10,7,8,9,1};
    quick(a,0,4);
    for(int i=0;i<5;i++) printf("%d ",a[i]);
}
