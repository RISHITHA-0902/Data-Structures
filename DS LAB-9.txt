#include<stdio.h>
int main(){
    int a[20],n,key,low=0,high,mid;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    scanf("%d",&key);

    high=n-1;
    while(low<=high){
        mid=(low+high)/2;
        if(a[mid]==key){ printf("Found"); break;}
        else if(key>a[mid]) low=mid+1;
        else high=mid-1;
    }
}
